# 이것은 간단한 함수입니다.

    """
    두 개의 숫자를 더하는 함수

    return num1 + num2

# 이것은 클래스입니다.

    def __init__(self, name, age):
        self.name = name
        self.age = age

    def greet(self):
        print(f"안녕하세요! 제 이름은 {self.name}이고, 제 나이는 {self.age}살입니다.")

# 이것은 인스턴스입니다.
person1 = Person("John", 30)
